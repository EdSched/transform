Supabase テーブル作成 SQL（初回のみ・大学院用）
CREATE TABLE admission_results (
  id         uuid primary key default gen_random_uuid(),
  univ       text not null,
  dept       text,
  spec       text,
  student    text,
  subject    text,
  era        text,
  jlpt       text,
  jlpt_score integer,
  eng_type   text,
  eng_score  integer,
  note       text,
  tier       integer,
  created_at timestamptz default now()
);
ALTER TABLE admission_results ENABLE ROW LEVEL SECURITY;
CREATE POLICY "public read" ON admission_results FOR SELECT USING (true);
CREATE POLICY "anon insert" ON admission_results FOR INSERT WITH CHECK (true);
CREATE POLICY "anon delete" ON admission_results FOR DELETE USING (true);
CREATE POLICY "anon update" ON admission_results FOR UPDATE USING (true);
